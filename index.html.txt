<script>
let score = 0;
const frases = [
  "pega você aqui ó ❤️ 🥺",
  "esse é pra você minha neném 💖",
  "um coração = um beijinho rsrs",
  "VOCÊ é simplesmente a melhor namorada de toda a história 💕",
  "só mais um pouquinho neném 🥹💗"
];

function criarCoracao() {
  const heart = document.createElement("div");
  heart.classList.add("heart");
  heart.innerHTML = "❤️";

  heart.style.left = Math.random() * 100 + "vw";
  heart.style.animationDuration = (Math.random() * 2 + 2) + "s";

  heart.onclick = () => {
    score++;
    document.getElementById("score").innerText = "❤️ " + score;

    // mensagem fofa aleatória
    document.getElementById("msg").innerText =
      frases[Math.floor(Math.random() * frases.length)];

    heart.remove();

    if (score >= 20) {
      finalizar();
    }
  };

  document.body.appendChild(heart);
  setTimeout(() => heart.remove(), 5000);
}

setInterval(criarCoracao, 700);

function finalizar() {
  document.getElementById("final").style.display = "flex";
}

/* mover gatinho (toque ou mouse) */
document.addEventListener("mousemove", e => {
  document.getElementById("player").style.left = e.clientX + "px";
});

document.addEventListener("touchmove", e => {
  document.getElementById("player").style.left =
    e.touches[0].clientX + "px";
});
</script>
